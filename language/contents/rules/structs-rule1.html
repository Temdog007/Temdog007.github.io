All structs must implement an interface

        interface IFoo
        {
            int x;
        }
        void UseFoo(IFoo f) => {}

        struct Foo : IFoo
        {
            int x;
        }
        UseFoo(Foo());

        // Since no interface is explicitly stated, all interfaces defined in the project will be
        // matched against this struct to see which interfaces it implements. In this case, it will match with IFoo.
        struct Foo2
        {
            int x;
        }
        UseFoo(Foo2());

        interface IFoo2
        {
            string s;
        }

        // This will not automatically implement IFoo since there is an interface stated.
        struct Foo3 : IFoo2
        {
            int x;
            string s;
        }
        UseFoo(Foo3()); // This will give an error

        // This will automatically implement IFoo because of the @auto annonation
        @auto
        struct Foo4 : IFoo2
        {
            int x;
            string s;
        }
        UseFoo(Foo4()));

Reasoning

Functions cannot take structs as parameters. Only interfaces. So, structs must implement an interface to be usuable.
