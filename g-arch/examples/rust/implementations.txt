#[derive(Debug, PartialEq, Copy, Clone)]
pub struct GameState {
    pub state: u8,
    pub control_player_state: u8,
    pub draw_player_state: u8,
    pub move_player_state: u8,

    players: EntityArray<Player>,
    timers : EntityArray<TimedEntity>
}

#[derive(PartialEq)]
pub enum GameStateModification {
    NoMod,

    ChangeState(u8),
    ChangeMovePlayerState(u8),
    ChangeControlPlayserState(u8),
    ChangeDrawPlayserState(u8),

    AddPlayer(Player),

    RemoveEntity(usize),

    ChangeMovablePosition(usize, Position),
    ChangeMovableVelocity(usize, Velocity),

    ChangeDrawablePosition(usize, Position),
    ChangeDrawableTexture(usize, Texture),

    ChangeTimedTime(usize, Timer),
    
    ChangeControllableVelocity(usize, Velocity),
    ChangeControllableController(usize, Controller),

    Multiple(Vec<GameStateModification>)
}
