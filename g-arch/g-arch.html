G-Arch is a tool designed to help aid in game architecture structure. This tool assumes that an entity component system will be used.

A component is a set of data.

Example: A position component that represents an entity's position a 2D game will have two numbers to represent the X and Y coordinate.

JSON Example

                            {
                                "name" : "position",
                                "properties": [
                                    {
                                        "name": "x",
                                        "type" : "float"
                                    },
                                    {
                                        "name" : "y",
                                        "type" : "float"
                                    }
                                ]
                            }

YAML Example

                            name: position
                            properties:
                            - 
                                name: x
                                type : float
                            - 
                                name : y
                                type : float

A family is a set of components. If an entity belongs to a family, then it must have all of the components in the set.

Example: Members of the 'movable' family will have a position component and a velocity component

JSON Example

                            {
                                "name" : "movable",
                                "components": [
                                    "position",
                                    "velocity"
                                ]
                            }

YAML Example

                            name : movable
                            components:
                                - position
                                - velocity

An entity is a set of components that belongs to one or more families. The families that the entity belongs to defines which components it has.

Example: The 'player' entity is a member of the 'movable', 'controllable', and 'drawable' families.

JSON Example

                            {
                                "name" : "player",
                                "families": [
                                    "movable",
                                    "controllable",
                                    "drawable"
                                ]
                            }

YAML Example

                            name : player
                            families:
                                - movable
                                - controllable
                                - drawable

A system is a function that one or more families as parameters. Every entity that belongs to a family that the function needs will be passed to this function.

Example: The 'movePlayer' system takes the 'moveable' family as a parameter.

JSON Example

                            {
                                "name" : "movePlayer",
                                "parameters" : [
                                    "movable"
                                ]
                            }

YAML Example

                            name : movePlayer
                            parameters:
                                - movable

//Components are structs
public struct Position {
    public readonly float x;
    public readonly float y;
}
public struct Velocity {
    public readonly float x;
    public readonly float y;
}
public struct Controller {
    public readonly string controllerType;
}
public struct Texture{
    public readonly string textureFilePath;
}

//Families are interfaces
public interface Movable {
    Position Position {get;}
    Velocity Velocity {get;}
}
public interface Drawable {
    Position Position{get;}
    Texture Texture{get;}
}
public interface Controllable {
    Velocity Velocity{get;}
    Controller Controller {get;}
}

// Entities are classes
public class Player : Movable, Drawable, Controllable {
    Position Position {get; set;}
    Velocity Velocity {get; set;}
    Texture Texture{get; set;}
    Controller Controller {get; set;}
}

// System manager will contain all of the systems
public interface SystemManager
{
    // Systems are pure functions. The arguments are based on the families it takes as parameters. A list is always passed for the families. 
    SystemReturnValue<Movable> MovePlayer(SystemState state, IReadOnlyList<Movable> movable);
    SystemReturnValue<Drawable> DrawPlayer(SystemState state, IReadOnlyList<Drawable> drawable);
    SystemReturnValue<Controlable> ControlPlayer(SystemState state, IReadOnlyList<Controlable> controllable);
}
//Each function will return the new state of the system and any changed components.
public struct SystemReturnValue<T> {
    // New system state
    public readonly SystemState state;

    // Item1 is the index of where the component was in the list     
    // Item2 is the new component
    public readonly IEnumerable<Tuple<int, T>> changes;
}
// A enum will be created and passed to every system function.
public enum SystemState {
    On,
    Off
}
